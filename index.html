        let currentView = 'main';

        let currentCategory = null;


        function showMainPage() {

            currentView = 'main';

            currentCategory = null;

            document.getElementById('subtitle').style.display = 'block';

            document.getElementById('category-name').style.display = 'none';

            renderGrid();

        }


        function showCategoryPage(categoryKey) {

            currentView = 'category';

            currentCategory = categoryKey;

            document.getElementById('subtitle').style.display = 'none';

            const categoryName = document.getElementById('category-name');

            categoryName.textContent = data[categoryKey].name;

            categoryName.style.display = 'block';

            categoryName.style.color = data[categoryKey].gradient.match(/#[0-9a-f]{6}/i)[0];

            renderGrid();

        }


        function renderGrid() {

            if (!data || loading) return;

            

            const grid = document.getElementById('grid');

            grid.innerHTML = '';


            if (currentView === 'main') {

                // Show featured article from each category

                Object.keys(data).forEach(categoryKey => {

                    const category = data[categoryKey];

                    const featuredArticle = category.articles.find(a => a.featured);

                    

                    const item = document.createElement('div');

                    item.className = 'grid-item';

                    item.style.background = category.gradient;

                    

                    item.innerHTML = `

                        <div class="overlay">

                            <div class="category" onclick="event.stopPropagation(); showCategoryPage('${categoryKey}')">${category.name}</div>

                            <div class="title">${featuredArticle.title}</div>

                        </div>

                    `;

                    

                    item.onclick = () => window.open(featuredArticle.url, '_blank');

                    grid.appendChild(item);

                });

            } else {

                // Show all articles in category

                const category = data[currentCategory];

                category.articles.forEach(article => {

                    const item = document.createElement('div');

                    item.className = 'grid-item';

                    if (article.featured) item.classList.add('featured');

                    item.style.background = category.gradient;

                    

                    item.innerHTML = `

                        <div class="overlay">

                            ${article.featured ? '<div class="featured-badge">Featured on Main</div>' : ''}

                            <div class="title">${article.title}</div>

                        </div>

                    `;

                    

                    item.onclick = () => window.open(article.url, '_blank');

                    grid.appendChild(item);

                });

            }

        }
